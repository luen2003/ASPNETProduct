CREATE DATABASE IndentityServerDB;
USE IdentityServerDB;
CREATE TABLE MD.Users (
    I BIGINT IDENTITY PRIMARY KEY,
    C NVARCHAR(100) NOT NULL, -- UserName
    N NVARCHAR(255),          -- FullName
    SHORTN NVARCHAR(100),
    PWD NVARCHAR(255),        -- Password (hash)
    UTYPE INT,
    PWDEXPIREDAY INT,
    LASTPWDCHANGED DATETIME,
    COMPANY BIGINT,
    POS BIGINT,
    TEL NVARCHAR(50),
    MOBILE NVARCHAR(50),
    ADDR NVARCHAR(255),
    MAILBOX NVARCHAR(255),
    CHATID NVARCHAR(255),
    SYSLOGININFO INT,
    SysU INT,
    SysD DATETIME,
    SysV INT,
    SysS INT
);
